def defaultMat(filePath, cameraRef="camProjShape", colorSpace="Raw", ignoreCsFileRules=False):
    
    matName = filePath.split("/")
    matName = matName[-1]
    matName = matName.split(".")
    matName = matName[0] + "Mat"
    
    shader = cmds.shadingNode("surfaceShader", asShader=True, name=matName)
    sg = cmds.sets(renderable=True, noSurfaceShader=True, empty=True, name=matName + "SG")
    cmds.connectAttr(shader + ".outColor", sg + ".surfaceShader", force=True)
    
    p2dtxt = cmds.shadingNode("place2dTexture", asUtility=True, name= matName + "Place2dTexture")
    file = cmds.shadingNode("file", asTexture=True, name= matName + "File")
    
    p2dtxtFileConnections = [
        ("outUV", "uvCoord"),
        ("outUvFilterSize", "uvFilterSize"),
        ("vertexCameraOne", "vertexCameraOne"),
        ("vertexUvOne", "vertexUvOne"),
        ("vertexUvThree", "vertexUvThree"),
        ("vertexUvTwo", "vertexUvTwo"),
        ("coverage", "coverage"),
        ("mirrorU", "mirrorU"),
        ("mirrorV", "mirrorV"),
        ("noiseUV", "noiseUV"),
        ("offset", "offset"),
        ("repeatUV", "repeatUV"),
        ("rotateFrame", "rotateFrame"),
        ("stagger", "stagger"),
        ("translateFrame", "translateFrame"),
        ("wrapU", "wrapU"),
        ("wrapV", "wrapV")
    ]

    for connect in p2dtxtFileConnections:
        cmds.connectAttr(p2dtxt + "." + connect[0], file + "." + connect[1], force=True)
    
    projection = cmds.shadingNode("projection", asUtility=True, name= matName + "Proj")
    cmds.connectAttr(file + ".outColor", projection + ".image", force=True)
    
    reverse = cmds.shadingNode("reverse", asUtility=True, name= matName + "Reverse")
    cmds.connectAttr(file + ".outAlpha", reverse + ".inputX", force=True)
    cmds.connectAttr(file + ".outAlpha", reverse + ".inputY", force=True)
    cmds.connectAttr(file + ".outAlpha", reverse + ".inputZ", force=True)
    cmds.connectAttr(reverse + ".output", projection + ".transparency", force=True)
    
    cmds.connectAttr(projection + ".outColor", shader + ".outColor", force=True)
    cmds.connectAttr(projection + ".outTransparency", shader + ".outTransparency", force=True)
        
    cmds.setAttr(file + ".fileTextureName", filePath, type="string")
    cmds.setAttr(file + ".cs", colorSpace, type="string")
    if ignoreCsFileRules:
        cmds.setAttr(file + ".ignoreColorSpaceFileRules", 1)
        
    cmds.setAttr(projection + ".projType", 8)
    cmds.connectAttr(cameraRef + ".message", projection + ".linkedCamera", force=True)
    cmds.setAttr(projection + ".fitType", 2)
    
    return sg


def assignDefaultMat(cameraRef="camProjShape", colorSpace="Raw", ignoreCsFileRules=False):

    filePath = cmds.fileDialog2(caption="image path", fileMode=1, fileFilter="Images (*.png *.jpg *.tif *.exr)")
    if filePath:
        filePath = filePath[0]

        geo = cmds.ls(selection=True)
        if not geo:
            cmds.warning("No geometry selected.")
            return
    
        sgMat = defaultMat(filePath, cameraRef=cameraRef, colorSpace=colorSpace, ignoreCsFileRules=ignoreCsFileRules)
        cmds.sets(geo, edit=True, forceElement=sgMat)
        
    
assignDefaultMat()
