def change_image_path(newPath):
    selection = cmds.ls(selection=True)
    for node in selection:
        path = cmds.getAttr(fnode + ".fileTextureName")
        path = path.split("/")
        path = newPath + path[-1]
        cmds.setAttr(fnode + ".fileTextureName", path, type="string")


change_image_path("../sourceimages/layers/")
